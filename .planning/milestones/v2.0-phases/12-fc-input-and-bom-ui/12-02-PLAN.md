```typescript
export type FCNetworkBOM = {
  fabricASwitches: number        // Fabric A switch count
  fabricBSwitches: number        // Fabric B switch count (always === fabricASwitches)
  hostPortsPerFabric: number
  storagePortsPerFabric: number
  islPortsPerFabric: number
  podLicensesRequired: number    // REQUIRED top-level line item (never optional)
  fcOpticsCount: number          // FC SFP optics total
  islCables: number              // ISL cable count
  fanInRatio: number             // host/storage fan-in (Broadcom 7:1 max → severity badge)
  islOversubscriptionRatio: number // informational — show separately
  violations: FCConstraintViolation[]
  input: FCSizingInput
}

export type FCConstraintViolation =
  | { code: 'FC_PORT_SATURATION'; requiredPorts: number; availablePorts: number }
  | { code: 'FC_OVERSUBSCRIPTION_EXCEEDED'; ratio: number; maxRatio: number }
  | { code: 'FC_ISL_UNDERPROVISIONED'; islsAvailable: number; islsRequired: number }
```

```typescript
  }},
})

// fanInRatio <= 3 → 'optimal'

  }
  if (v.code === 'FC_OVERSUBSCRIPTION_EXCEEDED') { ... }
  if (v.code === 'FC_ISL_UNDERPROVISIONED') { ... }
  return null
}
```
