**6. `src/features/export/exportFCPdf.ts`:**

```typescript
import type { FCNetworkBOM } from '@/domain/schemas/fc-bom'
import React from 'react'

export async function generateFCPdfBlob(
  bom: FCNetworkBOM,
  pngFabricA?: string,
  pngFabricB?: string,
): Promise<Blob> {
  const [{ pdf }, { FCNetStackDocument }] = await Promise.all([
    import('@react-pdf/renderer'),
    import('./pdf/FCNetStackDocument'),
  ])
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  const element = React.createElement(FCNetStackDocument, { bom, pngFabricA, pngFabricB }) as unknown as React.ReactElement<any>
  return pdf(element).toBlob()
}
```

```typescript
const handlePdfExport = async () => {
  setPdfGenerating(true)
  setPdfError(null)
  try {
    if (mode === 'fc') {
      if (!fcBom) return
      const pngA = getLastFCTopologyPng('A') ?? undefined
      const pngB = getLastFCTopologyPng('B') ?? undefined
      const blob = await generateFCPdfBlob(fcBom, pngA, pngB)
      const url = URL.createObjectURL(blob)
      const a = document.createElement('a')
      a.href = url
      a.download = 'netstack-fc-report.pdf'
      a.click()
      URL.revokeObjectURL(url)
      return
    }
    if (!bom) return
    const png = getLastTopologyPng()
    const blob = await generatePdfBlob(bom, png ?? undefined)
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a')
    a.href = url
    a.download = 'netstack-report.pdf'
    a.click()
    URL.revokeObjectURL(url)
  } catch (err) {
    setPdfError(err instanceof Error ? err.message : String(err))
  } finally {
    setPdfGenerating(false)
  }
}
```
