    ```typescript
    export function buildTopologyGraph(bom: NetworkBOM, canvasWidth = 1200): TopologyGraphResult {
      const xSpacing = (count: number) => canvasWidth / (count + 1)
      // Compute spine, leaf, OOB, rack nodes with deterministic x positions
      // Compute edges: leaf-spine (each leaf connects to each spine via uplinkPorts),
      //   server-leaf (one edge per rack to each of its 2 leafs),
      //   server-oob (one edge per rack to its OOB),
      //   VLT (one dashed edge between each leaf pair)
      // Return { nodes, edges }
    }
    ```
    Implementation details:
    - Spine nodes: `bom.spineSwitches` nodes at y=80, evenly spaced
    - Leaf nodes: `bom.leafSwitches` nodes at y=240, evenly spaced. Each pair (2i, 2i+1) belongs to rack i
    - OOB nodes: `bom.oobSwitches` nodes at y=240 + 60 (offset from leafs), one per rack
    - Rack nodes: `bom.racks` nodes at y=420
    - Spine node data: `{ model: 'S5232F-ON', role: 'spine', usedPorts: bom.leafSwitches, totalPorts: 32 }`

    - OOB node data: `{ model: 'S3248T-ON', role: 'oob', usedPorts: bom.input.serversPerRack + 2, totalPorts: 48 }`
    - Edge types: leaf-spine edges connect each leaf to ALL spines (one edge per uplink port per leaf, but simplified to one edge per leaf-spine pair). Server-leaf edges: one from rack-i to leaf-2i and leaf-2i+1. VLT: dashed edge from leaf-2i to leaf-2i+1.

    Export async function that uses html-to-image toPng with getNodesBounds + getViewportForBounds from @xyflow/react:
    ```typescript
    import { getNodesBounds, getViewportForBounds } from '@xyflow/react'
    import type { ReactFlowInstance } from '@xyflow/react'
    import { toPng } from 'html-to-image'

    export async function captureTopologyPng(
      rfInstance: ReactFlowInstance,
      bgColor: string
    ): Promise<string> {
      const IMAGE_W = 800, IMAGE_H = 500
      const bounds = getNodesBounds(rfInstance.getNodes())
      const viewport = getViewportForBounds(bounds, IMAGE_W, IMAGE_H, 0.5, 2, 0.1)
      const el = document.querySelector<HTMLElement>('.react-flow__viewport')
      if (!el) throw new Error('ReactFlow viewport not found')
      return toPng(el, {
        backgroundColor: bgColor,
        width: IMAGE_W,
        height: IMAGE_H,
        style: {
          width: `${IMAGE_W}px`,
          height: `${IMAGE_H}px`,
          transform: `translate(${viewport.x}px,${viewport.y}px) scale(${viewport.zoom})`,
        },
      })
    }
    ```
