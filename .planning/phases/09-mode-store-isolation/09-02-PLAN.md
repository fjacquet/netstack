```typescript
import type { PersistStorage, StorageValue } from 'zustand/middleware'

// Lazy adapter: reads window.localStorage at call time, not module load time
const lazyLocalStorage: PersistStorage<InputState> = {
  getItem: (name): StorageValue<InputState> | null => {
    try {
      const str = window.localStorage.getItem(name)
      if (str === null) return null
      return JSON.parse(str) as StorageValue<InputState>
    } catch { return null }
  },
  setItem: (name, value): void => {
    try { window.localStorage.setItem(name, JSON.stringify(value)) } catch {}
  },
  removeItem: (name): void => {
    try { window.localStorage.removeItem(name) } catch {}
  },
}

export const useInputStore = create<InputState>()(
  persist((set) => ({
    input: DEFAULT_INPUT,
    setInput: (partial) => set((state) => ({ input: { ...state.input, ...partial } })),
    resetInput: () => set({ input: DEFAULT_INPUT }),
  }), {
    name: 'netstack-input',   // ← FC store MUST use 'netstack-fc-input'
    version: 5,               // ← FC store uses version: 1 (new key, no legacy)
    storage: lazyLocalStorage,
    merge: (persisted, current) => { ... }
  })
)
```

From src/store/resultStore.ts (verbatim subscription pattern — replicate exactly):

```typescript
import { create } from 'zustand'
import { calculateBOM } from '@/domain/engine/sizing'
import type { NetworkBOM, ConstraintViolation } from '@/domain/schemas/bom'
import { useInputStore } from './inputStore'

interface ResultState {
  bom: NetworkBOM | null
  violations: ConstraintViolation[]
}

export const useResultStore = create<ResultState>()(() => ({ bom: null, violations: [] }))

// Module-level subscription — OUTSIDE React lifecycle
useInputStore.subscribe((state) => {
  try {
    const bom = calculateBOM(state.input)
    useResultStore.setState({ bom, violations: bom.violations })
  } catch {}
})

// Initial computation on module load
const initialInput = useInputStore.getState().input
try {
  const initialBom = calculateBOM(initialInput)
  useResultStore.setState({ bom: initialBom, violations: initialBom.violations })
} catch {}
```
